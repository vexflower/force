package engine;

import environment.RendererManager;
import hardware.Display;
import lang.GeomMath;
import lang.Mat4;
import renderer.MasterRenderer;
import ui.Panel;
import ui.scene.Scene3D;

public class Main {
    static void main(String[] args) {
        Display.createDisplay(0, 1280, 720);
        Display.setShowFPSTitle(true);
        MasterRenderer.setRenderer();

        Scene3D scene = new Scene(1280, 720)
        Display.setContentPane(scene);

        int texture = TextureLoader.loadTexture("brat");
        Entity ent = new Entity(mesh: Mesh.CUBE, texture: texture);
        ent.setPosition(0, 0, -2f);
        ent.moveRotate(0f, 200f, 0f, -1f (infinite movement)); // 200 degrees per second

        scene.addEntity(ent);

        // 2. A standard UI Panel (Static size)
        // panel should already call init() by itself btw...
        Panel myPanel = new Panel(300, 300);
        myPanel.setBackgroundColor(0.8f, 0.1f, 0.1f, 0.7f);
        myPanel.setPosition(50, 50);
        scene.add(myPanel);

        // Run the completely generic engine
        GameEngine.start();
    }
}
